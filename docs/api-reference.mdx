### Counter

Interactive counter component with customizable step size.

#### Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| None | - | - | Counter manages its own state |

#### State

```tsx
interface CounterState {
  count: number      // Current counter value
  step: number       // Increment/decrement step size
}
```

#### Methods

| Method | Parameters | Description |
|--------|------------|-------------|
| `increment()` | None | Increases count by step value |
| `decrement()` | None | Decreases count by step value |
| `reset()` | None | Resets count to 0 |

#### Example

```tsx
import Counter from './components/Counter'

function App() {
  return (
    <div>
      <Counter />
    </div>
  )
}
```

## üé£ Custom Hooks

### useLocalStorage

Store and retrieve data from localStorage with React state synchronization.

```tsx
function useLocalStorage<T>(key: string, initialValue: T) {
  const [storedValue, setStoredValue] = useState<T>(() => {
    try {
      const item = window.localStorage.getItem(key)
      return item ? JSON.parse(item) : initialValue
    } catch (error) {
      return initialValue
    }
  })

  const setValue = (value: T | ((val: T) => T)) => {
    try {
      const valueToStore = value instanceof Function ? value(storedValue) : value
      setStoredValue(valueToStore)
      window.localStorage.setItem(key, JSON.stringify(valueToStore))
    } catch (error) {
      console.log(error)
    }
  }

  return [storedValue, setValue] as const
}
```

#### Usage

```tsx
const [count, setCount] = useLocalStorage('count', 0)
const [todos, setTodos] = useLocalStorage<Todo[]>('todos', [])
```

### useAsync

Handle async operations with loading and error states.

```tsx
function useAsync<T>(asyncFunction: () => Promise<T>, dependencies: any[] = []) {
  const [data, setData] = useState<T | null>(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)

  useEffect(() => {
    let cancelled = false
    
    setLoading(true)
    setError(null)
    
    asyncFunction()
      .then(result => {
        if (!cancelled) {
          setData(result)
          setLoading(false)
        }
      })
      .catch(err => {
        if (!cancelled) {
          setError(err.message)
          setLoading(false)
        }
      })

    return () => {
      cancelled = true
    }
  }, dependencies)

  return { data, loading, error }
}
```

#### Usage

```tsx
const { data: users, loading, error } = useAsync(fetchUsers, [])
```

### formatDate

Format dates for display in components.

```tsx
function formatDate(date: Date | string): string {
  const d = new Date(date)
  return d.toLocaleDateString('en-US', {
    year: 'numeric',
    month: 'short',
    day: 'numeric'
  })
}
```

#### Usage

```tsx

// Output: "Jan 15, 2024"
```

### classNames

Conditionally join class names together.

```tsx
function classNames(...classes: (string | undefined | null | false)[]): string {
  return classes.filter(Boolean).join(' ')
}
```

#### Usage

```tsx
const buttonClass = classNames(
  'btn',
  isPrimary && 'btn-primary',
  isDisabled && 'btn-disabled'
)
```

### debounce

Debounce function calls to improve performance.

```tsx
function debounce<T extends (...args: any[]) => void>(
  func: T,
  wait: number
): (...args: Parameters<T>) => void {
  let timeout: NodeJS.Timeout
  
  return (...args: Parameters<T>) => {
    clearTimeout(timeout)
    timeout = setTimeout(() => func(...args), wait)
  }
}
```

#### Usage

```tsx
const debouncedSearch = debounce((query: string) => {
  // Perform search
}, 300)
```
